;	RA1 = D2 Led
;	RA2 = D3 Led
;	RA3 = D4 Led
;	RA4 = D5 Led
;	RA5 = D6 Led
;	RB0-3 = BCD value of display digit
;	RB4 = DP Display 0 = On
;	RB5 = Stat	0 = On
;	RB6 = aux	1 = On
;	RC0 = ID1
;	RC1 = ID2
;	RC3 = ID3
;	RC4 = ID4
;	RC5 = ID5

start


	clrf	flag1	
	clrf	fail
	banksel	0		;set bank 0
	clrf	porta		;clear port output latches
	clrf	portb
	clrf	portc	
    bsf	status,rp0 	;bank1
	movlw   b'00000000'	; set port B as output
   	movwf	trisb  
	movlw	b'00000000' 
	movwf	trisa		;set port A 0 - 7 to output
	movlw	b'00000111'
	movwf	adcon1		;TEMP - REQUIRED FOR A0-3 AS OUTPUT
	movlw	b'11100011'
	movwf	trisc		;set port C 0,1,3,4,5 as input
	banksel	0		;set bank0

;	call 	init_ad		RESET RA0 (TRISA)
	call	init_tmr21	;for .5 sec timer
	call	wait05		;delay 1/2 sec
	bsf	portb,5		;stat led off

	call	clr_disp
	call	tst_display2	;show 0 - 3 on each display
	call	init_tmr2	;for no_data monitor
	call 	init_uart
	call	init_tmr1

	bsf		intcon,7	;enable all interupts
	bsf		intcon,6	;Peripheral int enable
	bcf		pir1,rcif

	bcf		porta,2		;TEMP
	bcf 	pir1,tmr2if
;#
	clrf	buffin
	clrf	buffout

	call	setid

main
	btfsc	flag1,tim_fl	
	call 	pollmon		;flag got set by timer1 interupt
	btfsc pir1,tmr2if		;timer 2 flag
	call	no_data		;timer was not reset by valid data in
	btfsc	fail,6		;has fail been incremented up to 0x40? (4 Secs){8/7/08}
	call	data_fail
	btfsc	pir1,rcif		;check for data in buffer
	call	do_rcv		;recieved data at serial port
	goto 	main

read_port
	btfsc	rcsta,ferr	;chk for framing error
	goto	rcv_error1
	btfsc	rcsta,oerr	;rcv buffer overun error
	goto	rcv_error2
	movfw	rcreg
	return

rcv_error1
	bcf	rcsta,cren	;disable rcv to clear error
	bsf	rcsta,cren
	movfw	rcreg
	bcf	porta,5
	return
rcv_error2
	bcf	rcsta,cren	;disable rcv to clear error
	bsf	rcsta,cren
	movfw	rcreg
	bcf	porta,5
	return

pollmon
	;Called here by Timer 1 interupt - every 16? seconds
	;turn the fan off, clear the display 
	bcf	flag1,tim_fl
	bcf	porta,2
	call	clr_disp	;in case of error
	return

no_data
	incf	fail,f
	bcf 	pir1,tmr2if
	return

data_fail
	call	clr_disp
	bcf	porta,5		;led off
	return

clr_disp
	;no data in, clear the displays
	call	disp_bl			;data to blank display
	clrf	dig_sel
	call	wr_disp
	movlw	0x10
	movwf	dig_sel
	call	wr_disp
	movlw	0x20
	movwf	dig_sel
	call	wr_disp
	movlw	0x30
	movwf	dig_sel
	call	wr_disp	
	clrf	fail
	bcf	porta,5			;led off
	bsf	porta,4			;DP off
	bcf	porta,0			;trig output off ){8/7/08}
	return

tst_char

	comf	var2,f	
	incf	var2,f		;rcvd value = complement +1

;#### to here

	clrf	tmr1l
	clrf	tmr1h			;reset timer1 (stop timeout while using display) 		

chz
	clrf	tmr2		;received any char, reset the timer
	bsf		porta,5		;Led On
	clrf	fail		;reset Fail counter
	return


chx		;received some other char (From polling cut judges)
	bcf	flag1,digit
	bcf	flag1,disp
	clrf	tmr2		;received any char, reset the timer
	bsf	porta,5		;Led On
	clrf	fail		;reset Fail counter
	return

setid	;read id bits and set display id, board id
	movlw	01
	movwf	boardid		;preset to board 1

	movfw	portc


	goto	??		; bits 0,1,3,4,5 all 0, invalid
	btfsc	temp,1
	goto	id1		;id 1 = 1
	movlw	b'00000001'
	movwf	dispid
	goto	iddone

id1	btfsc	temp,2
	goto	id2		;id 2 = 1
	movlw	b'00000010'
	movwf	dispid
	goto	iddone

id2	movlw	b'00000100'	;setid to display 3
	movwf	dispid
	bcf	temp,1
	bcf	temp,2
	rrf 	temp,f		move id bits 3-5 to 0-3
	rrf	temp,f
	rrf	temp,f
	tstf
	skpnz			
	goto	iddone		id = 0, use board id = 1	
	decf	temp,f
	skpnz
	rlf	boardid,f


iddone

wr_disp
;requires dig_sel = display # (4514 data in), 0 = D1, 10 = D2, 20 = D3, 30 = D4
;         dig_dat1..4 = led state

	movfw	dig_sel
	movwf	portc				;leds 1 - 8
	movfw	dig_dat1
	movwf	portb
	bcf		porta,latch
	bsf		porta,latch		;latch data

	movlw	04
	addwf	dig_sel,w
	movwf	dig_sel
	movwf	portc				;leds 9 - 16
	movfw	dig_dat2
	movwf	portb
	bcf		porta,latch
	bsf		porta,latch		;latch data

	movlw	04
	addwf	dig_sel,w
	movwf	dig_sel
	movwf	portc				;leds 17 - 24
	movfw	dig_dat3
	movwf	portb
	bcf		porta,latch
	bsf		porta,latch		;latch data

	movlw	04
	addwf	dig_sel,w
	movwf	dig_sel
	movwf	portc				;leds 25 - 28
	movfw	dig_dat4
	movwf	portb
	bcf		porta,latch
	bsf		porta,latch		;latch data
	return

wait05	;0.5 sec delay => timer2 x value in temp
	movlw	0x20	
	movwf	temp
w1	call 	wait_t2
	decfsz 	temp,f
	goto	w1	
	bcf		flag1,tmp
	return


wait_t2
	bcf		pir1,1	;clear flag  tmr2 = 65536uS (value of temp)
d1	btfss	pir1,1
	goto	d1
	bcf		pir1,1
	return

wait_1ms
	bcf		pir1,1
  	bsf		status,rp0 	;bank1
	movlw 	0x39		;3/5 WAS 38
	movwf	pr2			;timer2 period reg
	banksel	0
b1	btfss	pir1,1
	goto	b1
	bcf		pir1,1
	return

	movlw	b'01001001'	;a/d on, ch1
	movwf	adcon0
	call 	delay1		;setup delay
	bsf		adcon0,2	;enable a/d
	goto	wait_ad

get_ad0  ;** return a/d result in w
	movlw	b'01000001'	;a/d on, ch0
	movwf	adcon0
	call 	delay1		;setup delay
	bsf		adcon0,2	;enable a/d
wait_ad
	btfsc	adcon0,2	; wait for done bit clear
	goto	wait_ad

init_uart
	bsf	status,5
	movlw	d'51'		;was 22 for 2400 baud
	movwf	0x19		;set baud rate 1200
	movlw	b'00100000'
	movwf	0x18		;enable uart xmit, 8bit, async	
	bcf	status,5
	movlw	b'10010000'
	movwf	0x18		;enable rx/tx port
	return

init_ad
	bsf		status,5
	movlw	b'00000101'	;sel ch 0,1 vref ch3
	movwf	0x1f		; =9f, adcon1
	bcf		status,5
	clrf	adres
	return

init_tmr1
;#	movlw	b'00101111'	;timer 1, prescale 4, Ext clk
	movlw	b'00111111'	;timer 1, prescale 8, Ext clk
;	movlw	b'00100101'	;timer 1, prescale 4, INT clk
	movwf	t1con
	clrf	tmr1l
	clrf	tmr1h
	bsf		status,5
	bsf		pie1,0		;enable timer1 interupt
	bcf		status,5
	return

init_tmr2
	movlw	b'01111110'		;prescale16, postscale 16, tmr on
	movwf	t2con			
	movlw	0xff			; = 65536uS to TMR2IF flag
	bsf		status,5
	movwf	pr2
	bcf		status,5
	return

init_tmr21
	movlw	b'01111110'		;prescale16, postscale 16, tmr on
	movwf	t2con			
	movlw	0x39			; = 14.6ms to TMR2IF flag
	bsf		status,5
	movwf	pr2
	bcf		status,5
	return

delay1	; about 10uS  (for a/d) 
	movlw	10		
	movwf	temp
sd
	decfsz	temp,1
	goto	sd
	return

delay2	; temp	 ABOUT 1MS
	movlw	0xff		
	movwf	temp
zz
	decfsz	temp,1
	goto	zz
	return



do_irq
	bsf		flag1,tim_fl	
	bcf		pir1,0		; clr int flag
	retfie	;return
